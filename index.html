import React, { useState, useEffect } from 'react';
import { CodeEditor, VideoCall, Chat } from '@hackhub/components';

const App = () => {
  const [code, setCode] = useState('');
  const [collaborators, setCollaborators] = useState([]);

  useEffect(() => {
    // Connect to collaborative session
    const socket = new WebSocket('wss://api.hackhub.dev/collab');

    socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      if (data.type === 'code_update') {
        setCode(data.code);
      }
    };

    return () => socket.close();
  }, []);

  const handleCodeChange = (newCode) => {
    setCode(newCode);
    // Broadcast change to collaborators
    broadcastCodeUpdate(newCode);
  };

  return (
    <div className="hackhub-container">
      <CodeEditor
        value={code}
        onChange={handleCodeChange}
        language="javascript"
        collaborators={collaborators}
      />
    </div>
  );
};

export default App;

About HackHub

Revolutionizing collaborative development since 2023

Our Story

HackHub was founded in 2023 by a team of developers who were frustrated with the limitations of existing collaborative coding tools. We wanted a platform that truly understood the needs of modern development teams.

What started as an internal tool for our own remote team quickly grew into a full-fledged platform that now serves thousands of developers worldwide.

Our mission is simple: to make collaborative development seamless, efficient, and accessible to everyone.

Our Vision

We believe that the future of software development is collaborative. Great code is built by teams, not individuals, and we're building the tools to make that collaboration effortless.

Frequently Asked Questions

How do I invite team members to a session?

You can invite team members by clicking the "Share" button in the editor and sending them the unique room link, or by entering their email addresses directly.

What programming languages are supported?

HackHub supports 50+ programming languages including JavaScript, Python, Java, C++, Ruby, Go, Rust, TypeScript, and many more.

Is my code secure on HackHub?

Yes, all code is encrypted end-to-end and stored securely. We're SOC 2 compliant and offer enterprise-grade security features.

Can I use HackHub for free?

Yes, HackHub offers a free tier with basic features. You can upgrade to premium plans for advanced features and team collaboration tools.
